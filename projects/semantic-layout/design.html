JavaScript Concepts Projects

1. Fundamentals

Basic concepts of JavaScript programming:

1.1 Variables and Data Types


// Variable declaration
let number = 5;
const PI = 3.14159;
var oldVar = "Hello";

// Data types
let string = "This is a string";
let boolean = true;
let array = [1, 2, 3];
let object = { key: "value" };

console.log(number, PI, oldVar, string, boolean, array, object);

1.2 Operators and Expressions


// Arithmetic operators
let sum = 5 + 3;
let difference = 10 - 4;
let product = 6 * 7;
let quotient = 20 / 4;

// Comparison operators
let isEqual = 5 === 5;
let isNotEqual = 10 !== 8;
let isGreater = 15 > 10;

console.log(sum, difference, product, quotient, isEqual, isNotEqual, isGreater);

1.3 Control Structures


// If-else statement
let age = 18;
if (age >= 18) {
    console.log("You are an adult");
} else {
    console.log("You are a minor");
}

// For loop
for (let i = 0; i < 5; i++) {
    console.log("Iteration " + i);
}

// While loop
let count = 0;
while (count < 3) {
    console.log("Count: " + count);
    count++;
}

2. Functions

Working with functions in JavaScript:

2.1 Function Declaration and Expression


// Function declaration
function greet(name) {
    return "Hello, " + name + "!";
}

// Function expression
const multiply = function(a, b) {
    return a * b;
};

console.log(greet("Alice"));
console.log(multiply(4, 5));

2.2 Arrow Functions


// Arrow function
const square = (x) => x * x;

// Arrow function with multiple parameters
const add = (a, b) => a + b;

console.log(square(5));
console.log(add(3, 7));

2.3 Higher-Order Functions


// Higher-order function
function operateOnNumbers(a, b, operation) {
    return operation(a, b);
}

// Using the higher-order function
const result = operateOnNumbers(10, 5, (x, y) => x - y);
console.log(result);

3. Objects and Arrays

Working with objects and arrays in JavaScript:

3.1 Object Creation and Manipulation


// Object creation
const person = {
    name: "John",
    age: 30,
    hobbies: ["reading", "music", "sports"]
};

// Accessing and modifying object properties
console.log(person.name);
person.age = 31;
person.occupation = "Developer";

console.log(person);

3.2 Array Methods


// Array creation
const fruits = ["apple", "banana", "orange"];

// Array methods
fruits.push("grape");
fruits.pop();
const slicedFruits = fruits.slice(1, 3);
const indexOfBanana = fruits.indexOf("banana");

console.log(fruits, slicedFruits, indexOfBanana);

3.3 Destructuring


// Object destructuring
const { name, age } = person;

// Array destructuring
const [firstFruit, secondFruit] = fruits;

console.log(name, age, firstFruit, secondFruit);
